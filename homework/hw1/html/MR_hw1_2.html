box

figure
a = zeros(256,256);
a(size(a,1)*3/8+1 : size(a,1)*5/8, size(a,2)*3/8+1 : size(a,2)*5/8) = 1;
subplot(1,3,1)
imagesc(a)
colormap(gray); axis image
title('a')
A= fft2(a);
Aprime = fftshift(A);
Mag = abs(Aprime);
Ph = angle(Aprime);
subplot(1,3,2)
imagesc(log(Mag+1))
colormap(gray); axis image
title(' ln( abs(A) + 1)');
subplot(1,3,3)
imagesc(Ph)
colormap(gray); axis image
title(' phase(A) ');
